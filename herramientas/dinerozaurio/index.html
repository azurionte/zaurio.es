import { supabase, signInWithGoogle, signOut } from './supabase.js'; import { state, resetStateButKeepSession } from './state.js'; import * as db from './db.js'; import * as ui from './ui.js'; import { ymNow, addMonths } from './utils.js'; async function bootstrap(){ ui.bindStaticEvents({ onGenerateForecast:()=>ui.renderForecast(), onGoogleLogin:async()=>{ await signInWithGoogle(); }, onLogout:async()=>{ await signOut(); }, onSavePlanMeta:savePlanMeta, onReload:loadEverything }); const { data:{ session } } = await supabase.auth.getSession(); state.session=session; ui.setAuthUI(session); if(session) await loadEverything(); supabase.auth.onAuthStateChange(async(_event, sessionNow)=>{ state.session=sessionNow; ui.setAuthUI(sessionNow); if(sessionNow) await loadEverything(); else { resetStateButKeepSession(); window.location.reload(); } }); document.getElementById('planNameInput').addEventListener('change',savePlanMeta); document.getElementById('initialReserveInput').addEventListener('change',savePlanMeta); document.getElementById('defaultStartMonthInput').addEventListener('change',savePlanMeta); document.getElementById('forecastFrom').addEventListener('change',()=>{ state.forecast.from=document.getElementById('forecastFrom').value; }); document.getElementById('forecastTo').addEventListener('change',()=>{ state.forecast.to=document.getElementById('forecastTo').value; }); document.getElementById('forecastGoal1').addEventListener('change',()=>{ state.forecast.goal1Id=document.getElementById('forecastGoal1').value; }); document.getElementById('forecastGoal2').addEventListener('change',()=>{ state.forecast.goal2Id=document.getElementById('forecastGoal2').value; }); } async function loadEverything(){ const session=state.session; if(!session) return; await db.ensureProfile(session.user); await db.loadOrCreateDefaultPlan(session.user.id); await db.loadPlanData(state.plan.id); if(!state.forecast.from) state.forecast.from=state.plan.default_start_month||ymNow(); if(!state.forecast.to) state.forecast.to=addMonths(state.forecast.from,12); ui.renderAll(); ui.renderForecast(); } async function savePlanMeta(){ if(!state.plan) return; state.plan.name=document.getElementById('planNameInput').value.trim()||'Plan principal'; state.plan.initial_reserve=Number(document.getElementById('initialReserveInput').value||0); state.plan.default_start_month=document.getElementById('defaultStartMonthInput').value||ymNow(); await db.savePlanMeta(); ui.renderAll(); ui.renderForecast(); } bootstrap();
